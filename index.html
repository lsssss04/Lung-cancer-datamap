    <script>

        // 1. DATA INFRASTRUCTURE

        const categories = [

            { id: "NSCLC", color: "#4cc9f0", count: 40 },

            { id: "Metformin/Metabolic", color: "#7209b7", count: 25 },

            { id: "Immunotherapy", color: "#f72585", count: 35 },

            { id: "SCLC", color: "#4895ef", count: 20 }

        ];


        let nodes = [];

        let links = [];


        // Generate synthetic historical data

        categories.forEach(cat => {

            const rootId = `root-${cat.id}`;

            nodes.push({ id: rootId, label: cat.id, group: cat.id, isRoot: true, year: 2015, radius: 15 });

            

            for(let i=0; i < cat.count; i++) {

                const year = 2015 + Math.floor(Math.random() * 12);

                const id = `${cat.id}-${i}`;

                nodes.push({

                    id: id,

                    label: "",

                    group: cat.id,

                    year: year,

                    radius: 4 + Math.random() * 6,

                    title: `Study ${cat.id} #${i + 100}`,

                    abs: `Clinical analysis of ${cat.id} pathways. Year: ${year}. Correlation: 0.82.`

                });

                links.push({ source: rootId, target: id, year: year });

            }

        });


        // 2. D3 SYSTEMS SETUP

        const width = window.innerWidth;

        const height = window.innerHeight;

        const svg = d3.select("#map-canvas").append("svg").attr("width", width).attr("height", height);

        const g = svg.append("g");


        svg.call(d3.zoom().scaleExtent([0.1, 8]).on("zoom", (e) => g.attr("transform", e.transform)));


        const simulation = d3.forceSimulation()

            .force("link", d3.forceLink().id(d => d.id).distance(50).strength(0.4))

            .force("charge", d3.forceManyBody().strength(-150))

            .force("center", d3.forceCenter(width / 2, height / 2))

            .force("collision", d3.forceCollide().radius(d => d.radius + 5));


        const linkLayer = g.append("g").attr("class", "links");

        const nodeLayer = g.append("g").attr("class", "nodes");

        const labelLayer = g.append("g").attr("class", "labels");


        // 3. RENDER ENGINE

        function update(yearFilter, searchStr) {

            const filteredNodes = nodes.filter(n => 

                n.year <= yearFilter && 

                (searchStr === "" || n.group.toLowerCase().includes(searchStr) || n.title?.toLowerCase().includes(searchStr))

            );


            const nodeIds = new Set(filteredNodes.map(n => n.id));

            const filteredLinks = links.filter(l => 

                nodeIds.has(typeof l.source === 'object' ? l.source.id : l.source) &&

                nodeIds.has(typeof l.target === 'object' ? l.target.id : l.target)

            );


            document.getElementById('count-nodes').innerText = filteredNodes.length;


            // Nodes

            const n = nodeLayer.selectAll("circle").data(filteredNodes, d => d.id);

            n.exit().remove();

            const nEnter = n.enter().append("circle")

                .attr("class", "node")

                .attr("r", d => d.radius)

                .attr("fill", d => categories.find(c => c.id === d.group).color)

                .style("opacity", d => d.isRoot ? 1 : 0.8)

                .on("click", (e, d) => showDetails(d))

                .call(d3.drag().on("start", dragStart).on("drag", dragging).on("end", dragEnd));


            // Links

            const l = linkLayer.selectAll("line").data(filteredLinks, d => `${d.source.id || d.source}-${d.target.id || d.target}`);

            l.exit().remove();

            l.enter().append("line").attr("class", "link").attr("stroke", "#1e293b");


            // Labels

            const lab = labelLayer.selectAll("text").data(filteredNodes.filter(d => d.isRoot), d => d.id);

            lab.exit().remove();

            lab.enter().append("text")

                .attr("class", "label")

                .text(d => d.label)

                .attr("dy", -20)

                .attr("text-anchor", "middle");


            simulation.nodes(filteredNodes);

            simulation.force("link").links(filteredLinks);

            simulation.alpha(0.3).restart();

        }


        // 4. INTERACTION LOGIC

        function showDetails(d) {

            const container = document.getElementById('details');

            container.innerHTML = `

                <h2 style="font-size: 1rem; color: var(--accent); margin-bottom: 5px;">${d.title || d.label}</h2>

                <div style="margin-bottom: 10px;">

                    <span class="tag">${d.group}</span>

                    <span class="tag">${d.year}</span>

                </div>

                <p style="font-size: 0.8rem; line-height: 1.4; color: #cbd5e1;">${d.abs || 'Core hub for research cluster.'}</p>

                <button style="margin-top:10px; background:var(--accent); color:#000; border:none; padding:5px 12px; border-radius:4px; font-weight:bold; cursor:pointer; font-size:0.7rem;">ANALYZE DATA</button>

            `;

        }


        document.getElementById('year-slider').oninput = function() {

            document.getElementById('year-val').innerText = this.value;

            update(this.value, document.getElementById('search').value.toLowerCase());

        };


        document.getElementById('search').oninput = function() {

            update(document.getElementById('year-slider').value, this.value.toLowerCase());

        };


        simulation.on("tick", () => {

            linkLayer.selectAll("line").attr("x1", d => d.source.x).attr("y1", d => d.source.y).attr("x2", d => d.target.x).attr("y2", d => d.target.y);

            nodeLayer.selectAll("circle").attr("cx", d => d.x).attr("cy", d => d.y);

            labelLayer.selectAll("text").attr("x", d => d.x).attr("y", d => d.y);

        });


        function dragStart(e) { if (!e.active) simulation.alphaTarget(0.3).restart(); e.subject.fx = e.subject.x; e.subject.fy = e.subject.y; }

        function dragging(e) { e.subject.fx = e.x; e.subject.fy = e.y; }

        function dragEnd(e) { if (!e.active) simulation.alphaTarget(0); e.subject.fx = null; e.subject.fy = null; }


        update(2026, "");

        window.onresize = () => { svg.attr("width", window.innerWidth).attr("height", window.innerHeight); };

    </script>
